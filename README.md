```python
# Data structures you'll actually use in AI systems

index = defaultdict(list)
for doc_id, token in documents:
```

```python
# ── Consuming APIs (OpenAI style) ─────────────────────────────────────
import httpx
from tenacity import retry, wait_exponential, stop_after_attempt

@retry(wait=wait_exponential(min=1, max=60), stop=stop_after_attempt(5))
async def call_api(prompt: str) -> str:
    async with httpx.AsyncClient(timeout=30.0) as client:
        response = await client.post(
            "https://api.openai.com/v1/chat/completions",
            headers={"Authorization": f"Bearer {API_KEY}"},
            json={"model": "gpt-4o", "messages": [{"role": "user", "content": prompt}]}
        )
        response.raise_for_status()
        return response.json()["choices"][0]["message"]["content"]

# ── Building APIs with FastAPI ─────────────────────────────────────────
from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel

app = FastAPI(title="AI Product API", version="1.0.0")

class ChatRequest(BaseModel):
    message: str
    session_id: str
    model: str = "gpt-4o"

class ChatResponse(BaseModel):
    reply: str
    tokens_used: int
    latency_ms: float

@app.post("/chat", response_model=ChatResponse)
async def chat_endpoint(req: ChatRequest):
    start = time.time()
    reply = await call_llm(req.message, req.model)
    return ChatResponse(
        reply=reply,
        tokens_used=count_tokens(reply),
        latency_ms=(time.time() - start) * 1000
    )
```

## 2.1 NumPy

### 📖 Concept
NumPy is the backbone of all numerical computing in Python. Neural networks, embeddings, and matrix operations all rely on NumPy arrays (or tensors, which are NumPy's GPU counterparts). Understanding it deeply makes PyTorch feel natural.

### 🔑 Key Topics
- [ ] ndarray creation, shapes, dtypes
- [ ] Indexing, slicing, fancy indexing
- [ ] Broadcasting rules
- [ ] Vectorized operations (no Python loops!)
- [ ] Linear algebra: `np.dot`, `np.linalg`
- [ ] Random number generation

```python
import numpy as np

# Embedding math — what happens inside vector databases
embedding_a = np.random.randn(1536)   # OpenAI embedding
embedding_b = np.random.randn(1536)

# Cosine similarity (without sklearn)
def cosine_sim(a: np.ndarray, b: np.ndarray) -> float:
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

# Broadcasting — apply operation across all embeddings at once
embeddings = np.random.randn(1000, 1536)   # 1000 documents
query = np.random.randn(1536)

# Dot product of query with ALL embeddings in one line (no loop!)
scores = embeddings @ query                 # shape: (1000,)
top_5_indices = np.argsort(scores)[-5:]    # top 5 most similar
```

```python
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.metrics import classification_report, roc_auc_score
from sklearn.model_selection import cross_val_score
import optuna

# Full ML training loop
full_pipeline = Pipeline([
    ("preprocessor", preprocessor),
    ("model", RandomForestClassifier(n_estimators=200, random_state=42))
])

full_pipeline.fit(X_train, y_train)
y_pred = full_pipeline.predict(X_test)
y_prob = full_pipeline.predict_proba(X_test)[:, 1]

print(classification_report(y_test, y_pred))
print(f"AUC-ROC: {roc_auc_score(y_test, y_prob):.4f}")

# Hyperparameter tuning with Optuna
def objective(trial):
    params = {
        "n_estimators": trial.suggest_int("n_estimators", 50, 300),
        "max_depth": trial.suggest_int("max_depth", 3, 20),
        "min_samples_split": trial.suggest_int("min_samples_split", 2, 20)
    }
    model = RandomForestClassifier(**params, random_state=42)
    return cross_val_score(model, X_train, y_train, cv=5, scoring="roc_auc").mean()

study = optuna.create_study(direction="maximize")
study.optimize(objective, n_trials=50)
print("Best params:", study.best_params)
```

## 3.1 Neural Networks Fundamentals

### 📖 Concept
Neural networks are the mathematical engines behind every LLM, image model, and speech model. Understanding forward passes, backpropagation, and loss optimization is what separates AI engineers who just call APIs from those who can fine-tune, debug, and architect AI systems.

```python
import torch
import torch.nn as nn
import torch.optim as optim

# Build a neural network from scratch in PyTorch
class TextClassifier(nn.Module):
    def __init__(self, vocab_size: int, embed_dim: int, hidden_dim: int, num_classes: int):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=0)
        self.encoder = nn.Sequential(
            nn.Linear(embed_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
            nn.GELU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.GELU(),
        )
        self.classifier = nn.Linear(hidden_dim // 2, num_classes)

    def forward(self, input_ids: torch.Tensor) -> torch.Tensor:
        x = self.embedding(input_ids)      # (B, seq, embed)
        x = x.mean(dim=1)                  # mean pooling → (B, embed)
        x = self.encoder(x)                # (B, hidden//2)
        return self.classifier(x)          # (B, num_classes)

# Training loop
model = TextClassifier(vocab_size=30000, embed_dim=256, hidden_dim=512, num_classes=5)
optimizer = optim.AdamW(model.parameters(), lr=3e-4, weight_decay=0.01)
scheduler = optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=100)
criterion = nn.CrossEntropyLoss()

for epoch in range(num_epochs):
    for batch_ids, batch_labels in dataloader:
        optimizer.zero_grad()
        logits = model(batch_ids)
        loss = criterion(logits, batch_labels)
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        optimizer.step()
    scheduler.step()
```

## 4.2 Embeddings

### 📖 Concept
Embeddings convert text (or images, audio) into dense numerical vectors where semantic similarity = vector proximity. They're the bridge between human language and machine computation — foundational to semantic search, RAG, recommendation systems, and clustering.

### 🔑 Key Topics
- [ ] Word embeddings: Word2Vec, GloVe
- [ ] Sentence embeddings: Sentence-BERT, `sentence-transformers`
- [ ] OpenAI embeddings API (`text-embedding-3-large`)
- [ ] Similarity metrics: cosine, dot product, Euclidean
- [ ] Embedding use cases: search, clustering, classification, deduplication
- [ ] Matryoshka embeddings (truncatable dimensions)
- [ ] Embedding model benchmarks (MTEB leaderboard)

```python
from sentence_transformers import SentenceTransformer
import numpy as np

model = SentenceTransformer("BAAI/bge-large-en-v1.5")

# Embed a knowledge base
documents = [
    "Python is a high-level programming language",
    "Machine learning is a subset of artificial intelligence",
    "Neural networks are inspired by the human brain",
    "Transformers use self-attention mechanisms"
]
doc_embeddings = model.encode(documents, normalize_embeddings=True)  # shape: (4, 1024)

# Semantic search
query = "How do large language models work?"
query_embedding = model.encode([query], normalize_embeddings=True)

# Cosine similarity via dot product (normalized vectors)
scores = doc_embeddings @ query_embedding.T   # shape: (4, 1)
ranked = np.argsort(scores.squeeze())[::-1]

for i in ranked[:3]:
    print(f"Score: {scores[i][0]:.3f} | {documents[i]}")
```

## 4.3 Fine-Tuning Models

### 📖 Concept
Fine-tuning adapts a pre-trained model to your specific task or domain. Instead of training from scratch (millions of $), you update a small fraction of parameters on your data. This is how companies build specialized AI: customer support bots, legal document analyzers, medical coding assistants.

### 🔑 Key Topics
- [ ] Full fine-tuning vs. PEFT (Parameter-Efficient Fine-Tuning)
- [ ] LoRA and QLoRA — the most practical fine-tuning methods
- [ ] Supervised Fine-Tuning (SFT) with instruction data
- [ ] DPO — aligning models with preference data
- [ ] Dataset formats: Alpaca, ShareGPT, ChatML
- [ ] Evaluation during fine-tuning
- [ ] Tools: `trl`, `unsloth`, `axolotl`

```python
from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig
from peft import LoraConfig, get_peft_model, TaskType
from trl import SFTTrainer, SFTConfig
from datasets import Dataset

# QLoRA — fine-tune 7B model on consumer GPU
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_quant_type="nf4",
    bnb_4bit_compute_dtype=torch.bfloat16,
    bnb_4bit_use_double_quant=True
)

model = AutoModelForCausalLM.from_pretrained(
    "mistralai/Mistral-7B-Instruct-v0.3",

        output_dir="./finetuned-model",

        gradient_accumulation_steps=8, learning_rate=2e-4,
        bf16=True, logging_steps=10, max_seq_length=2048
    ),
    train_dataset=dataset
)
```

## 5.2 LangChain

### 📖 Concept
LangChain is the most widely-used framework for building LLM-powered applications. It provides chains, agents, memory, tools, and integrations for 100+ LLMs and databases. It's the "Rails of AI" — batteries included for building fast.

### 🔑 Key Topics
- [ ] LCEL (LangChain Expression Language) — pipe operator (`|`)
- [ ] Chat models: `ChatOpenAI`, `ChatAnthropic`, `ChatOllama`
- [ ] Prompt templates: `ChatPromptTemplate`, `MessagesPlaceholder`
- [ ] Output parsers: `StrOutputParser`, `JsonOutputParser`, Pydantic
- [ ] Document loaders: PDF, web, YouTube, CSV, etc.
- [ ] Text splitters and chunking strategies
- [ ] Retrievers and vector store integration
- [ ] Conversation memory with `RunnableWithMessageHistory`
- [ ] LangSmith for tracing and evaluation

```python
from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough, RunnableWithMessageHistory
from langchain_community.chat_message_histories import ChatMessageHistory
from langchain_community.vectorstores import Chroma
from langchain_community.document_loaders import PyPDFLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter

llm = ChatOpenAI(model="gpt-4o", temperature=0, streaming=True)

# ── RAG chain with LCEL ───────────────────────────────────────────────
loader = PyPDFLoader("product_manual.pdf")
docs = loader.load()

splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200)
chunks = splitter.split_documents(docs)

vectorstore = Chroma.from_documents(chunks, OpenAIEmbeddings())
retriever = vectorstore.as_retriever(search_kwargs={"k": 5})

rag_prompt = ChatPromptTemplate.from_template("""
You are a helpful product assistant. Answer based only on the provided context.
If you cannot answer from the context, say so.

Context: {context}
Question: {question}
""")

rag_chain = (
    {"context": retriever | (lambda docs: "\n\n".join(d.page_content for d in docs)),
     "question": RunnablePassthrough()}
    | rag_prompt | llm | StrOutputParser()
)

# Stream the answer
for chunk in rag_chain.stream("How do I reset the device?"):
    print(chunk, end="", flush=True)

# ── Conversation with persistent memory ──────────────────────────────
store = {}

def get_history(session_id: str) -> ChatMessageHistory:
    if session_id not in store:
        store[session_id] = ChatMessageHistory()
    return store[session_id]

chat_prompt = ChatPromptTemplate.from_messages([
```

```python
from fastapi import FastAPI, BackgroundTasks
from fastapi.responses import StreamingResponse
import redis.asyncio as aioredis
import asyncio
import json
import uuid

app = FastAPI()
redis_client = aioredis.from_url("redis://localhost:6379")

# Async job submission (for long-running AI tasks)
@app.post("/ai/analyze-document")
async def analyze_document(doc_id: str, background_tasks: BackgroundTasks):
    job_id = str(uuid.uuid4())
    await redis_client.setex(f"job:{job_id}", 3600, json.dumps({"status": "queued"}))
    background_tasks.add_task(run_ai_analysis, job_id, doc_id)
    return {"job_id": job_id, "status": "queued"}

@app.get("/ai/jobs/{job_id}")
async def get_job_status(job_id: str):
    result = await redis_client.get(f"job:{job_id}")
    return json.loads(result) if result else {"status": "not_found"}

# Streaming endpoint (Server-Sent Events)
@app.post("/ai/chat/stream")
async def chat_stream(message: str):
    async def generate():
        async with AsyncOpenAI() as client:
            stream = await client.chat.completions.create(
                model="gpt-4o", stream=True,
                messages=[{"role": "user", "content": message}]
            )
            async for chunk in stream:
                if chunk.choices[0].delta.content:
                    yield f"data: {json.dumps({'text': chunk.choices[0].delta.content})}\n\n"
        yield "data: [DONE]\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream")
```

```python
    response = await call_next(request)

    duration = time.time() - start
    request_count.labels(model=model, endpoint=request.url.path).inc()
    request_latency.labels(model=model).observe(duration)
    active_requests.dec()
    return response

# Expose metrics endpoint
app.mount("/metrics", make_asgi_app())
```

## 8.2 User Experience for AI Products

### 📖 Concept
AI UX has unique challenges that standard UX principles don't fully address: latency, non-determinism, hallucinations, and building trust. Great AI UX makes the AI feel reliable, fast, and empowering — not scary or frustrating.

### 🔑 Key Topics
- [ ] **Perceived performance:** Streaming, skeleton loaders, progress indicators
- [ ] **Trust signals:** Sources/citations, confidence scores, "I don't know"
- [ ] **Error states:** Graceful degradation when AI fails or hallucinates
- [ ] **Prompt UX:** Guided input, prompt suggestions, examples
- [ ] **Feedback loops:** Thumbs up/down, correction flows, RLHF data collection
- [ ] **Explainability:** Show reasoning, highlight sources, audit trails
- [ ] **Safety UX:** Warning banners, guardrail messages
- [ ] **Onboarding:** First run experience, demo data, sample queries

### AI UX Principles

| Principle | Implementation |
|-----------|---------------|
| 🚀 **Stream everything** | Never make users stare at a loading spinner for >2s |
| 📍 **Cite sources** | Show where information came from — builds trust |
| 🎯 **Set expectations** | "This AI can make mistakes. Always verify important information." |
| 🔄 **Allow correction** | Let users edit AI outputs, rate responses, give feedback |
| 💡 **Guide the prompts** | Suggest what to ask — reduces blank-page anxiety |
| 🛡️ **Fail gracefully** | When AI can't help, say so clearly and offer alternatives |

## 8.3 Evaluation of AI Outputs

### 📖 Concept
"How good is our AI?" is the hardest question in AI product development. Without rigorous evaluation, you can't know if a model change helped or hurt, or if your product is actually solving the user's problem. Build an eval culture early.

### 🔑 Key Topics
- [ ] Unit evals: test specific AI behaviors with assertions
- [ ] LLM-as-judge: use GPT-4 to score GPT-3.5 outputs
- [ ] Human evaluation: rating scales, A/B tests
- [ ] Golden datasets: curated question-answer pairs
- [ ] Regression testing: catch regressions when models change
- [ ] Online metrics: thumbs up rate, retry rate, session length
- [ ] RAG-specific: RAGAS (faithfulness, relevancy, recall)

## 8.4 AI Safety and Reliability

### 📖 Concept
AI products that make it to production must be safe, reliable, and trustworthy. A single viral hallucination, inappropriate response, or prompt injection attack can destroy a product. Safety is a product feature, not an afterthought.

### 🔑 Key Topics
- [ ] Input guardrails: topic restrictions, PII detection, injection defense
- [ ] Output guardrails: toxicity detection, factual grounding
- [ ] Prompt injection prevention
- [ ] Hallucination mitigation: RAG grounding, self-check prompting
- [ ] Rate limiting to prevent abuse
- [ ] Jailbreak resistance techniques
- [ ] Audit logging for compliance
- [ ] Graceful degradation and fallback models

```python
# ── Layered safety guardrails ──────────────────────────────────────────
import re
from openai import OpenAI
from pydantic import BaseModel

client = OpenAI()

class SafetyCheck(BaseModel):
    is_safe: bool
    reason: str
    risk_level: str   # low, medium, high

SYSTEM_PROMPT = """
You are a helpful AI assistant for a legal research platform.
ONLY answer questions about legal research, case law, and legal procedures.
NEVER:
- Provide specific legal advice as if you are a licensed attorney
- Share personal information about real people
- Generate harmful, illegal, or unethical content
- Answer questions outside the legal domain

If asked about something outside your scope, politely redirect.
"""

def check_input_safety(user_input: str) -> SafetyCheck:
    """Screen user input before sending to main LLM"""
    result = client.beta.chat.completions.parse(
        model="gpt-4o-mini",
        messages=[{
            "role": "user",
            "content": f"Is this input safe for a legal research AI? Assess risks.\nInput: {user_input}"
        }],
        response_format=SafetyCheck
    )
    return result.choices[0].message.parsed

def sanitize_input(text: str) -> str:
    """Remove potential prompt injection patterns"""
    injection_patterns = [
        r"ignore previous instructions",
        r"forget everything",
        r"you are now",
        r"jailbreak",
        r"<\|.*?\|>"
    ]
    for pattern in injection_patterns:
        if re.search(pattern, text, re.IGNORECASE):
            raise ValueError("Potentially unsafe input detected")
    return text.strip()

async def safe_generate(user_message: str) -> str:
    # 1. Sanitize
    clean_input = sanitize_input(user_message)

    # 2. Safety check
    safety = check_input_safety(clean_input)
    if not safety.is_safe or safety.risk_level == "high":
        return f"I can't help with that. {safety.reason}"

    # 3. Generate with constrained system prompt
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": clean_input}
        ]
    )
    return response.choices[0].message.content
```
